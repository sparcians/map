156 void addValue(EnumType enum_val)
157 {
158 uint64_t val = static_cast<uint64_t>(enum_val);
159 ++total_;
160
161 if (val < lower_val_) {
162 ++ (*underflow_bin_);
163 }
164 else if (val > upper_val_) {
165 ++ (*overflow_bin_);
166 }
167 else {
168 uint32_t idx = (val - lower_val_) >> idx_shift_amount_;
169 ++ (bin_.at(idx));
170 }
