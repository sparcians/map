298 std::string getPath() const {
299 std::stack<const std::string*> names;
300 Node const * n = this;
301 while(n && n->getName().size() > 0){ // Stop at null parent, or root (which has no name)
302 names.push(&n->getName());
303 n = n->getParent();
304 }
305
306 std::stringstream ss;
307 if(names.size() > 0){
308 ss << *names.top();
309 names.pop();
310 while(names.size() > 0){
311 ss << '.' << *names.top();
312 names.pop();
313 }
314 }
315 return ss.str();
316 }

442 Node* getChild(const std::string& name) const {
443 if(TreeNode::hasWildcardCharacters(name)){
444 throw SpartaException("Cannot call ParameterTree::Node::getChild with a name that is a search pattern: \"")
445 << name << "\". addChild must be used instead";
446 }
447
448 // Always search in reverse-applied order to match on most recent changes first
449 auto itr = children_.rbegin();
450 for(; itr != children_.rend(); ++itr){
451 if(matches((*itr)->getName(), name)){
452 return itr->get();
453 }
454 }
455
456 return nullptr;
457 }

469 Node* create(const std::string& path, bool required) {
470 if(path.size() == 0){
471 return this;
472 }
473
474 size_t name_pos = 0;
475 std::string immediate_child_name;
476 immediate_child_name = TreeNode::getNextName(path, name_pos);
477 if(immediate_child_name.size() == 0){
478 // Cannot get parent like this
479 return nullptr; // TEMPORARY behavior. See docstring
480 //throw SpartaException("Virtual parameter path \"") << path
481 // << "\" is invalid because it contains an empty name (between two '.' "
482 // "characters). Parents cannot currently be referenced in the virtual parameter tree";
483 }
484
485 // Get a child of node if one exists with an exact match before any nodes
486 // with wilcards are encountered. If this node name includes a wildcard, a
487 // child will be returned IFF it the pattern string itself matches with the
488 // highest priority child in this node.
489 Node* child = getPriorityChildMatch(immediate_child_name);
490 if(!child){
491 child = addChild(immediate_child_name, required);
492 }
493
494 if(name_pos == std::string::npos){
495 return child;
496 }
497
498 const std::string remainder = path.substr(name_pos);
499 return child->create(remainder, required);
500 }

510 Node* getPriorityChildMatch(const std::string& name) const {
511 if(children_.size() == 0){
512 return nullptr;
513 }
514
515 // Becase the name is a pattern, it is impossible to math against other existing
516 // children which are patterns. Therefore, if a pattern child is encountered
517 // which does not exactly match the given name string, nullptr must be returned so
518 // that the caller appends a new node, whih will have the highest priority
519 const bool name_has_wildcard = TreeNode::hasWildcardCharacters(name);
520
521 // Always search in reverse-applied order to match on most recent changes first.
522 //
523 // This is an optimization to prevent nodes from being created needlessly. It tries
524 // to find a matching node while considering wildcards to avoid creating new nodes
525 // in most cases where it is possible.
526 //
527 // Read the rules below to understand what happens. In general, if we don't get an
528 // exact string match with an existing node before hitting another node which is
529 // an exclusive superset or incomplete subset of the 'name' argument's pattern (it
530 // may have no wildcards) then a new node will need to be added at the end of the
531 // children list to specify a value. An existing node cannot be overridden.
532 auto itr = children_.rbegin();
533 for(; itr != children_.rend(); ++itr){
534 if((*itr)->getName() == name){
535 // Found a node with an exact name match (no pattern matching) before
536 // hitting a pattern node. This node can be used to apply a new parameter
537 return itr->get();
538 }
539
540 if(name_has_wildcard){
541 if(TreeNode::hasWildcardCharacters((*itr)->getName())){
542 // Encountered a wildcard node whch was not an exact string match.
543 // No way to tell if the name pattern exactly matches this node's, so
544 // assume it doesn't
545 return nullptr;
546 }else if(matches(name, (*itr)->getName())){
547 // Node has not wildcards but matches the pattern in 'name'. Therfore,
548 // A new parameter with this name would affect this node and more. A new
549 // node will have to be added to override it.
550 return nullptr;
551 }else{
552 // Node has no wildcards and the name agument is a pattern which
553 // does not match it. Therefore, it can be ignored because this 'name'
554 // will not apply
555 continue;
556 }
557 }else if(matches((*itr)->getName(), name)){
558 // Encountered a wildcard node which matches on this name before hitting an
559 // exact match. Therefore, a new node must be created by the caller so that
560 // the parameter being set will affect a subset of this node's pattern
561 return nullptr;
562 }
563 }
564 // No matches or important mismatches
565 return nullptr;
566 }

571 Node* addChild(const std::string& name, bool required) {
572 sparta_assert(hasValue() == false,
573 "Cannot add a child to a virtual parameter tree node \"" << name_
574 << "\" since it already has a value: \"" << value_ << "\"");
575 children_.emplace_back(new Node(this, name));
576 if(required){
577 children_.back()->incRequired();
578 }
579 return (children_.back().get());
580 }

1027 void recursAppendTree_(const Node* ot) {
1028 // Inherit value. Never invalidate
1029 if(ot->hasValue()){
1030 setValue(ot->peekValue(), ot->getRequiredCount() > 0, ot->getOrigin());
1031 }
1032
1033 // Inherit user data.

1035 user_data_[ud_name] = ud_value;
1036 }
1037
1038 for(auto & child : ot->getChildren()){
1039 // TODO: copy required count instead of just boolean
1040 Node* c = create(child->getName(), child->getRequiredCount() > 0); // Create if needed
1041 c->recursAppendTree_(child);
1042 }

1389 template<typename NodeT>
1390 static NodeT* recursTryGet_(NodeT* node, const std::string& path,
1391 const std::string& match_name,
1392 size_t name_pos, const bool must_be_leaf)
1393 {
1394 sparta_assert(!TreeNode::hasWildcardCharacters(match_name),
1395 "Cannot attempt to read a node with a path containing wildcard "
1396 "characters. A specific node path must be used. Error in \""
1397 << match_name << "\" from \"" << path << "\"");
1398
1399 if(name_pos == std::string::npos){
1400 // End of the search.. No deeper
1401 NodeT* result = nullptr;
1402 NodeT* backup = nullptr; // First match (if is has no value)
1403 auto itr = node->getMatcherBegin();
1404 for(; itr != node->getMatcherEnd(); ++itr){
1405 if(itr.matches(match_name)){
1406 if(itr.get()->hasValue() || !must_be_leaf){
1407 itr.get()->incrementReadCount();
1408 if(!result){
1409 result = itr.get(); // Found it
1410 }
1411 }else if(!backup){
1412 backup = itr.get();
1413 }
1414 }
1415 }
1416 if(result){
1417 return result;
1418 }
1419 return backup; // No match here
1420 }
1421
1422 // Search deeper
1423 std::string immediate_child_name;
1424 immediate_child_name = TreeNode::getNextName(path, name_pos);
1425
1426 if(immediate_child_name.size() == 0){
1427 // Cannot get parent.
1428 throw SpartaException("Parameter ") << path
1429 << " is invalid because it contains an empty name (between two '.' "
1430 "characters). Parents cannot currently be refrenced in the parameter tree";
1431 }
1432
1433 NodeT* result = nullptr;
1434 auto itr = node->getMatcherBegin();
1435 for(; itr != node->getMatcherEnd(); ++itr){
1436 if(itr.matches(match_name)){
1437 NodeT* match = recursTryGet_(itr.get(), path, immediate_child_name, name_pos, must_be_leaf);
1438 if(match && (match->hasValue() || !must_be_leaf)) {
1439 match->incrementReadCount();
1440 if(result == nullptr){
1441 result = match;
1442 }
1443 }
1444 // Keep this match as result and continue iterating to mark all matching nodes as read
1445 }
1446 }
1447 return result;
1448 }

1453 bool recursGetIsRequired_(const Node* node,
1454 const std::string& path,
1455 const std::string& match_name,
1456 size_t name_pos,
1457 bool& found_match) const
1458 {
1459 found_match = false;
1460
1461 if(name_pos == std::string::npos){
1462 // End of the search.. No deeper
1463 auto itr = node->getMatcherBegin();
1464 for(; itr != node->getMatcherEnd(); ++itr){
1465 if(TreeNode::hasWildcardCharacters(match_name)
1466 ? itr->getName() == match_name // Exact name match if match_name has wildcards
1467 : itr.matches(match_name)) // Pattern match if match_name has no wildcards
1468 {
1469 found_match = true;
1470 return itr.get()->getRequiredCount() > 0;
1471 }
1472 }
1473
1474 return false; // dummy
1475 }
1476
1477 // Search deeper
1478 std::string immediate_child_name;
1479 immediate_child_name = TreeNode::getNextName(path, name_pos);
1480
1481 if(immediate_child_name.size() == 0){
1482 // Cannot get parent.
1483 throw SpartaException("Parameter ") << path
1484 << " is invalid because it contains an empty name (between two '.' "
1485 "characters). Parents cannot currently be refrenced in the parameter tree";
1486 }
1487
1488 auto itr = node->getMatcherBegin();
1489 for(; itr != node->getMatcherEnd(); ++itr){
1490 if(itr.matches(match_name)){
1491 const bool required = recursGetIsRequired_(itr.get(), path, immediate_child_name, name_pos, found_match);
1492 if(found_match){

1494 }
1495 // No match found. Keep going
1496 }
1497 }
