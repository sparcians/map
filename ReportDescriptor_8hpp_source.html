643 void push_back(const ReportDescriptor & rd) {
644 //Internal book-keeping is only for Python shell workflows.
645 //We also do not perform any book-keeping for descriptors
646 //whose dest_file is "1" (print to stdout). That is the
647 //only use case where duplicate dest_file's is actually
648 //okay. We will add the descriptor to the collection, but
649 //any attempts to access this descriptor later (i.e. to
650 //change its parameters or to disable it from Python) will
651 //throw an exception that the descriptor "1" cannot be found.
652 if (rd.dest_file != "1") {
653 const std::string desc_name = getDescriptorName_(rd);
654 auto iter = indices_by_descriptor_name_.find(desc_name);
655 if (iter != indices_by_descriptor_name_.end()) {
656 //Descriptor found by this name. Throw if it is already
657 //enabled. Users can only replace disabled descriptors.
658 if (rep_descs_.at(iter->second).isEnabled()) {
659 throw SpartaException("Report descriptor named '") <<
660 desc_name << "' already exists in this configuration";
661 }
662 }
663 indices_by_descriptor_name_[desc_name] = rep_descs_.size();
664 }
665
666 //This incoming 'rd' variable could be a reference to a
667 //previously disabled descriptor:
668 //
669 // >>> rd = report_config.descriptors.foo_csv
670 // >>> report_config.removeReport('foo_csv')
671 // >>> report_config.addReport(rd)
672 //
673 //This would still count as an enabled descriptor.
674 rep_descs_.push_back(rd);
675 rep_descs_.back().enabled_ = true;
676 }

710 bool contains(const std::string & desc_name) const {
711 auto iter = indices_by_descriptor_name_.find(desc_name);
712 if (iter == indices_by_descriptor_name_.end()) {
713 //The name may have been given to us with a period, not
714 //an underscore, separating the file stem and the extension
715 const auto replaced_desc_name = replaceDotsWithUnderscores_(desc_name);
716 iter = indices_by_descriptor_name_.find(replaced_desc_name);
717 }
718
719 //If this name is not even found in our map, we do not
720 //contain this descriptor
721 if (iter == indices_by_descriptor_name_.end()) {
722 return false;
723 }
724
725 //But even if the descriptor is physcially here, we have
726 //to also ask it if is is still enabled
727 return rep_descs_.at(iter->second).isEnabled();
728 }

785 size_t getDescriptorIndexByName_(const std::string & desc_name) const {
786 auto iter = indices_by_descriptor_name_.find(desc_name);
787 if (iter == indices_by_descriptor_name_.end()) {
788 //The name may have been given to us with a period, not
789 //an underscore, separating the file stem and the extension
790 const auto replaced_desc_name = replaceDotsWithUnderscores_(desc_name);
791 iter = indices_by_descriptor_name_.find(replaced_desc_name);
792
793 //If it still is not found, error out
794 if (iter == indices_by_descriptor_name_.end()) {
795 throw SpartaException(
796 "No descriptor named '") << desc_name << "' exists";
797 }
798 }
799
800 //The descriptor exists in this collection, but we still need
801 //to throw if it has been disabled, since we are not supposed
802 //to be using those descriptors for any reason anymore.
803 if (!rep_descs_.at(iter->second).isEnabled()) {
804 throw SpartaException("The descriptor named '") << desc_name
805 << "' has already been disabled";
806 }
807 return iter->second;
808 }
