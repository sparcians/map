153 {
154 auto option = feature_options_.tryGet(option_name);
155 if (option == nullptr || !option->hasValue()) {
156 return default_value;
157 }
158
159 const std::string & opt_value_str = option->getValue();

162 return static_cast<T>(opt_value);
